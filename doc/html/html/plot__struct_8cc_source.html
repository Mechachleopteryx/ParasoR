375 void Plot::MakePairMat(string structure)
376 {
377  vector<int> stack;
378  pairs = vector<int>(length+2, 0);
379  pairs[0] = length;
380  for (int i = 1; i <= length; i++) {
381  switch (structure[i-1]) {
382  case '(':
383  stack.push_back(i); break;
384  case ')':
385  if (stack.size() > 0) {
386  pairs[i] = stack[stack.size()-1];
387  pairs[pairs[i]] = i;
388  stack.pop_back();
389  }
390  case '.':
391  default:
392  pairs[i] = 0;
393  break;
394  }
395  }
396 }

398 int Plot::GetCoordinates(Vec& X, Vec& Y)
399 {
400  DOUBLE INIT_ANGLE=0.; /* initial bending angle */
401  DOUBLE INIT_X = 100.; /* coordinate of first digit */
402  DOUBLE INIT_Y = 100.; /* see above */
403  DOUBLE RADIUS = 15.;
404  angle = Vec(length+5);
405  GetLoopAngle(0, length+1, pairs);
406 
407  DOUBLE alpha = INIT_ANGLE;
408  X[0] = INIT_X;
409  Y[0] = INIT_Y;
410  for (int i = 1; i <= length; i++) {
411  X[i] = X[i-1]+RADIUS*cos(alpha);
412  Y[i] = Y[i-1]+RADIUS*sin(alpha);
413  alpha += M_PI-angle[i+1];
414  }
415  return length;
416 }

418 int Plot::GetNextPair(int count, int p, const vector<int>& pairs, vector<int>& remember)
419 {
420  int k = p, l = pairs[p];
421  int start_k = k, start_l = l;
422  count += 2;
423  remember.push_back(k);
424  remember.push_back(l);
425  int ladder = 0;
426  for (; ; k++, l--, ladder++) {
427  if (pairs[k] != l)
428  break;
429  }
430  int fill = ladder-2;
431  if (ladder >= 2) {
432  angle[start_k+1+fill] += M_PI/2.; /* Loop entries and */
433  angle[start_l-1-fill] += M_PI/2.; /* exits get an */
434  angle[start_k] += M_PI/2.; /* additional PI/2. */
435  angle[start_l] += M_PI/2.; /* Why ? (exercise) */
436  if (ladder > 2) {
437  for (; fill >= 1; fill--) {
438  angle[start_k+fill] = M_PI; /* fill in the angles */
439  angle[start_l-fill] = M_PI; /* for the backbone */
440  }
441  }
442  }
443  GetLoopAngle(k, l, pairs);
444  return count;
445 }

447 void Plot::GetLoopAngle(int i, int j, const vector<int>& pairs)
448  /* i, j are the positions AFTER the last pair of a stack; i.e
449  i-1 and j+1 are paired. */
450 {
451  int count = 2; /* counts the VERTICES of a loop polygon; that's
452  NOT necessarily the number of unpaired bases!
453  Upon entry the loop has already 2 vertices, namely
454  the pair i-1/j+1. */
455  int bubble = 0; /* bubble counts the unpaired digits in loops */
456 
457  vector<int> remember(1, 0);
458  for (int p = i, q = j+1; p != q; )
459  {
460  if (!pairs[p] || (p == 0))
461  p++, count++, bubble++;
462  else {
463  count = GetNextPair(count, p, pairs, remember);
464  p = pairs[p]+1;
465  }
466  if (p == q)
467  remember.push_back(q);
468  }
469  BendAngleLoop(max(i-1, 0), M_PI*(count-2)/(DOUBLE)count, remember); /* bending angle in loop polygon */
470 }
